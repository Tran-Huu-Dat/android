public class MainActivity extends AppCompatActivity {

private ActivityResultLauncher<Intent> launcher = registerForActivityResult(new ActivityResultContracts.StartActivityForResult(), new ActivityResultCallback<ActivityResult>() {
@Override
public void onActivityResult(ActivityResult result) {
if(result.getResultCode() == RESULT_OK && result.getData() != null){
String maSV = result.getData().getStringExtra("maSV");
String tenSV = result.getData().getStringExtra("tenSV");
String lop = result.getData().getStringExtra("lop");
sinhViens.add(new SinhVien(maSV, tenSV, lop));
loadlsvSinhVien();
writeFile();
}
}
});

private void findViews(){
lsvSinhVien = findViewById(R.id.lsvSinhVien);
btnAdd = findViewById(R.id.btnAdd);
}
private void addEvents(){
btnAdd.setOnClickListener(new View.OnClickListener() {
@Override
public void onClick(View v) {
Intent intent = new Intent(MainActivity.this, Them_SinhVien.class);
launcher.launch(intent);
}
});
}

public void writeFile(){
try{
FileOutputStream stream = openFileOutput("sinhvien.txt", MODE_PRIVATE);
OutputStreamWriter writer = new OutputStreamWriter(stream);
writer.write(sinhViens.size() + "\n");
for(SinhVien sv : sinhViens){
writer.write(sv.getMaSV() + ";" + sv.getTenSV() + ";" + sv.getLop() + "\n");
}
writer.close();
stream.close();
Toast.makeText(this, "Ghi file thÃ nh cÃ´ng.", Toast.LENGTH_LONG).show();
}
catch (Exception ex){
Toast.makeText(this,"Ghi file tháº¥t báº¡i: " + ex.toString(), Toast.LENGTH_LONG).show();
}
}

private void readFile(){
try{
FileInputStream stream = openFileInput("sinhvien.txt");
InputStreamReader streamReader = new InputStreamReader(stream);
BufferedReader reader = new BufferedReader(streamReader);
int n = Integer.parseInt(reader.readLine());
sinhViens.clear();
for(int i=0; i<n; i++){
String[] data = reader.readLine().split(";");
sinhViens.add(new SinhVien(data[0], data[1], data[2]));
}
reader.close();
streamReader.close();
stream.close();
Toast.makeText(this, "Ä á» c file thÃ nh cÃ´ng.", Toast.LENGTH_LONG).show();
}
catch (Exception ex){
Toast.makeText(this,"Ä á» c file tháº¥t báº¡i: " + ex.toString(), Toast.LENGTH_LONG).show();
}
}

package com.example.ontap_tuan11;

import android.content.Context;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.TextView;

import androidx.annotation.NonNull;
import androidx.annotation.Nullable;

import java.util.ArrayList;

public class SinhVienAdapter extends ArrayAdapter<SinhVien> {
private Context context;
private ArrayList<SinhVien> sinhViens;

public SinhVienAdapter(Context context, ArrayList<SinhVien> sinhViens) {
super(context, R.layout.sinh_vien_item, sinhViens);
this.context = context;
this.sinhViens = sinhViens;
}

@NonNull
@Override
public View getView(int position, @Nullable View convertView, @NonNull ViewGroup parent) {
if(convertView == null){
LayoutInflater inflater = LayoutInflater.from(context);
convertView = inflater.inflate(R.layout.sinh_vien_item, parent, false);

}

//Láº¥y dá»¯ liá»‡u
SinhVien sinhVien = sinhViens.get(position);

//Ã nh xáº¡
TextView tvwMaSV = convertView.findViewById(R.id.tvwMaSV);
TextView tvwHoTen = convertView.findViewById(R.id.tvwHoTen);
TextView tvwLop = convertView.findViewById(R.id.tvwLop);
Button btnDel = convertView.findViewById(R.id.btnDel);

//Set dá»¯ liá»‡u
tvwMaSV.setText(sinhVien.getMaSV());
tvwHoTen.setText(sinhVien.getTenSV());
tvwLop.setText(sinhVien.getLop());

btnDel.setOnClickListener(new View.OnClickListener() {
@Override
public void onClick(View v) {
sinhViens.remove(position);
notifyDataSetChanged();

// Ghi láº¡i file
((MainActivity) context).writeFile();
}
});

return convertView;
}
}
